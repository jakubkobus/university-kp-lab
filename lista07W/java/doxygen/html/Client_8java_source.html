101 private void sendRequest() {
102 String command = commandBox.getValue();
103 String type = typeBox.getValue();
104 String value = valueField.getText().trim();
105
106 if (!command.equals("draw") && value.isEmpty()) {
107 setStatus(false);
108 return;
109 }
110
111 // Walidacja wejścia dla int i double
112 if (!command.equals("draw")) {
113 if (type.equals("int")) {
114 try {
115 Integer.parseInt(value);
116 } catch (NumberFormatException e) {
117 setStatus(false);
118 return;
119 }
120 } else if (type.equals("double")) {
121 try {
122 Double.parseDouble(value);
123 } catch (NumberFormatException e) {
124 setStatus(false);
125 return;
126 }
127 }
128 }
129
130 String request = command + " " + type + (command.equals("draw") ? "" : " " + value);
131
132 // Czyści pole po wysłaniu
133 javafx.application.Platform.runLater(() -> valueField.clear());
134
135 Thread t = new Thread(() -> {
136 try (Socket socket = new Socket(SERVER_HOST, SERVER_PORT);
137 PrintWriter writer = new PrintWriter(socket.getOutputStream(), true);
138 ObjectInputStream in = new ObjectInputStream(socket.getInputStream())) {
139
140 writer.println(request);
141 writer.flush();
142
143 Response response = (Response) in.readObject();
144 boolean success = response.status == Response.Status.SUCCESS;
145 if (command.equals("draw")) {
146 setDrawArea(response.tree != null ? response.tree : "Drzewo puste");
147 }
148 setStatus(success);
149 } catch (IOException | ClassNotFoundException e) {
150 setStatus(false);
151 }
152 });
153
154 t.setDaemon(true);
155 t.start();
156 }
